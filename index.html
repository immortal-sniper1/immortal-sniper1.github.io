let levelInput = document.getElementById("level"); let tempInput = document.getElementById("temp"); let flowInput = document.getElementById("flow"); let powerInput = document.getElementById("power"); let fromFlowButton = document.getElementById("from_flow"); let fromPowerButton = document.getElementById("from_power"); let fuelValue = 0.59; fromFlowButton.addEventListener("click", function() { let temp = tempInput.value; let flow = flowInput.value; let efficiency = levelInput.value; let k = (fuelValue * efficiency * 100 * (temp - 15)) / (1000 - 15); powerInput.value = k * flow / 1000; }) fromPowerButton.addEventListener("click", function() { let temp = tempInput.value; let power = powerInput.value; let efficiency = levelInput.value; let k = (fuelValue * efficiency * 100 * (temp - 15)) / (1000 - 15); flowInput.value = power * 1000 / k; })
